TO FIND FIBONACCI SERIES OF NUMBERS

def fibonacci(number):

    if(number==0):
 
       return 0

    elif number==1:

        return 1

    else:

        return fibonacci(number-2)+fibonacci(number-1)
 
   number=int(input("Please enter the fibonacci number range="))

sum=0

for num in range(number):

     print(fibonacci(num),end='')
 
    sum=sum+fibonacci(num)
  
   print("\n The sum of fibonacci series numbers=%d",%sum)

TO FIND FACTORIAL VALUE OF GIVEN NUMBER
def factorial(n):

   if(n==0):

       return 0

   if n==1:

       return 1

   else:

        return n*factorial(n-1)

   n=int(input(" enter any number="))
  
 print(''factorial value of',n,'is':',factorial(n))

THE GIVEN STRING IS PALINDROME OR NOT

def pal(s):

  s=s.lower()

  if len(s)<=1:

    return True

  elif s[0]!=s[-1]:

    return False

  else:

    return pal(s[1:-1])

myinput=input("Enter a string:")

if pal(myinput):

     print(f"'{myinput}'is a palindrome")

else:

     print(f"'{myinput}'is not a palindrome.")

def gcd(a,b):

  if b==0:
